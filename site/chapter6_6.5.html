  12:   * @invariant _wellformed()
  13:   */
  14:  public class Stack implements IStack, Cloneable {
  15:   
  16:      /**
  17:       * Creates a stack with size of 200
  18:       */
  19:      public Stack(){
  20:          stackArray = new Object[200];
  21:          assert _wellformed();
  22:      }
  23:      
  24:      /**
  25:       * Creates a stack with size of  @param stackSize
  26:       * 
  27:       * @param stackSize Size of stack 
  28:       */    
  29:      public Stack(int stackSize){
  30:          stackArray = new Object[stackSize];
  31:          assert _wellformed();
  32:      }
  33:      
  34:   
  35:      
  36:      /**
  37:       * Adds a new object onto top of the stack. 
  38:       * 
  39:       * @pre item != null
  40:       * @post size() == size()@pre + 1
  41:       * @post item@pre == top()
  42:       * @param item Item to be pushed to the top of the stack. 
  43:       */
  44:      public void push(Object item) {
  45:          assert _wellformed();
  46:          assert item!=null;
  47:          int preSize = size();
  48:          stackArray[size()] = item;
  49:          size++;
  50:          assert size() == preSize + 1;
  51:          assert item.equals(top());
  52:      }
  53:   
  54:      
  55:      /**
  56:       * Removes the object at the top of the stack and returns the object. 
  57:       * 
  58:       * @pre size() > 0
  59:       * @post @result == top()@pre
  60:       * @post size() == size()@pre - 1
  61:       * @return the object at the top of the stack, null if 
  62:       *            this stack contains no elements
  63:       */
  64:      public Object pop() {
  65:          assert _wellformed();
  66:          assert size() > 0;
  67:          int preSize = size();
  68:          Object top = top();
  69:          Object item = null;
  70:          if(!isEmpty()){
  71:              item = stackArray[size()-1];
  72:              stackArray[size()-1] = null;
  73:              size--;
  74:              assert item.equals(top);
  75:              assert size() == preSize - 1;
  76:          }
  77:          return item;
  78:      }
  79:   
  80:      /**
  81:       * Returns the object at the top of the stack without removing it from 
  82:       * the stack
  83:       * 
  84:       * @pre size() > 0
  85:       * @post @nochange
  86:       * @return the object at the top of the stack, null if 
  87:       *            this stack contains no elements
  88:       */
  89:      public Object top() {
  90:          assert _wellformed();
  91:          assert size() > 0;
  92:          int preSize = size();
  93:          Object item = null;
  94:          if(!isEmpty())
  95:              item = stackArray[size()-1];
  96:          assert preSize == size();
  97:          return item;
  98:      }    
  99:   
 100:      /**
 101:       * Returns the number of objects currently in the stack.
 102:       * 
 103:       * @pre true
 104:       * @post @nochange
 105:       * @return number of items in the stack
 106:       */
 107:      public int size() {
 108:          assert _wellformed();
 109:          return size;
 110:      }
 111:   
 112:      /**
 113:       * Returns true if this collection contains no elements.
 114:       * This implementation returns size() == 0.
 115:       * 
 116:       * @pre true
 117:       * @post @nochange
 118:       * @return true if this stack contains no elements
 119:       */    
 120:      public boolean isEmpty() {
 121:          return size() == 0;
 122:      }    
 123:   
 124:      /**
 125:       * Compares this stack to the specified object. 
 126:       * The result is true if and only if the argument 
 127:       * is not null and is a stack object that 
 128:       * represents the same sequence of objects.
 129:       * @param other object to compare
 130:       * @return true it other stack is equal to this stack
 131:       */
 132:      public boolean equals(Object other){
 133:          if (this == other) return true;
 134:          if(other instanceof Stack){
 135:              Stack otherStack = (Stack)other;
 136:              if(otherStack.size() == this.size()){
 137:                  //Compare items at the same index,
 138:                  //Make sure they are equal
 139:                  for(int i=0; i < size(); i++){
 140:                      if(stackArray[i] != otherStack.stackArray[i])
 141:                          return false;
 142:                  }
 143:                  return true;
 144:              }
 145:          }    
 146:          return false;
 147:      }
 148:   
 149:      
 150:      /**
 151:       * Returns a clone of this stack. The copy will contain a 
 152:       * reference to a clone of the internal array, 
 153:       * not a reference to the original array object.
 154:       * @return a clone of this stack.
 155:       */
 156:      public Stack clone() throws CloneNotSupportedException{
 157:          Stack clone = (Stack)super.clone();
 158:          clone.stackArray = new Object[stackArray.length];
 159:          for(int i=0; i < size(); i++){
 160:              clone.stackArray[i] = stackArray[i];
 161:          }
 162:          clone.size = this.size;
 163:          return clone;        
 164:      }
 165:   
 166:      /**
 167:       * Invariants of the stack implementation
 168:       * @return true if stack is well formed
 169:       */
 170:      protected boolean _wellformed(){
 171:          if (stackArray == null) return false;        
 172:          return true;
 173:      }            
 174:      
 175:      //Internal array to hold stack items
 176:      private Object[] stackArray;
 177:      private int size = 0;
 178:      
 179:  }
