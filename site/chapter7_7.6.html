   1:  package ex76;
   2:   
   3:  class InsertionSortAlgorithm extends SortAlgorithm {
   4:    void sort(int a[]) {
   5:      for (int i = 0; i < a.length; i++ )
   6:        for (int j = i; j > 0; j--) {
   7:          if (a[j-1] > a[j])
   8:            swap(a, j-1, j);
   9:          pause();
  10:        }
  11:    }

  12:      g.setColor(Color.white);
  13:      g.fillRect(0, 0, d.width, d.height);
  14:      double f = d.height / (double) a.length;    
  15:   
  16:      int x = d.width - 3;
  17:      for (int i = a.length; --i >= 0; x -= 3) {
  18:        g.setColor(Color.BLACK);
  19:        g.drawLine(x, d.height -(int)(a[i] * f), x+3, d.height -(int)(a[i] * f));
  20:      }
  21:    }   
  22:  }
