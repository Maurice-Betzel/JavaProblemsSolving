  31:

  42:   
  43:      public void actionPerformed(ActionEvent evt)
  44:      {    
  45:          try{

  51:              }

  60:      private int calculate()
  61:      {    
  62:          int result = 1;
  63:          int val = Integer.parseInt(txtNum.getText());
  64:          while(val > 0){
  65:              result *= val--;
  66:          }
  67:          return result;
  68:      }
  69:      
  70:      private boolean isValidInput()

  72:          String val = txtNum.getText();

  75:          return true;
  76:      }
  77:  }
