   6:  public class DigitalClock extends Applet implements Runnable
   7:  {
   8:      protected Thread clockThread = null;
   9:      protected Font font = null;
  10:      protected Color color = null;
  11:      protected Font dateFont = null;
  12:      protected Color borderColor = null;
  13:   
  14:      String[] monthName = {"Jan", "Feb",
  15:              "Mar", "Apr", "May", "Jun", "Jul",
  16:              "Aug", "Sep", "Oct", "Nov",
  17:              "Dec"};
  18:   
  19:      String[] dayName = {"Sun", "Mon", "Tue",
  20:              "Wed", "Thu", "Fri", "Sat"};
  21:      
  22:      public void init()
  23:      {    
  24:          int fontSize = 20;
  25:          try{
  26:              fontSize = Integer.parseInt(getParameter("SIZE"));
  27:          }catch(Exception ex){}
  28:          String fontName = getParameter("FONT");
  29:          font = new Font(fontName, Font.BOLD, fontSize);
  30:          color = getColor("COLOR");
  31:          dateFont = new Font(fontName, Font.BOLD, (int)(fontSize*0.7) );
  32:          borderColor = getColor("BORDER_COLOR");
  33:      }
  34:      public void start()
  35:      {
  36:          if(clockThread == null){
  37:              clockThread = new Thread(this);
  38:              clockThread.start();
  39:          }
  40:      }
  41:      
  42:      public void stop()
  43:      {
  44:          clockThread = null;
  45:      }
  46:      
  47:      public void run(){
  48:          while(Thread.currentThread() == clockThread){            
  49:              repaint();            
  50:              try{
  51:                  Thread.currentThread();
  52:                  Thread.sleep(1000);
  53:              }catch (InterruptedException e) {
  54:                  // TODO: handle exception
  55:              }
  56:          }
  57:      }
  58:      
  59:      public void paint(Graphics g){
  60:          Calendar calendar = Calendar.getInstance();
  61:          int hour = calendar.get(Calendar.HOUR);
  62:          int minute = calendar.get(Calendar.MINUTE);
  63:          int second = calendar.get(Calendar.SECOND);
  64:          int day = calendar.get(Calendar.DAY_OF_MONTH);
  65:          int month = calendar.get(Calendar.MONTH);
  66:          int year = calendar.get(Calendar.YEAR );
  67:          int amPm = calendar.get(Calendar.AM_PM);
  68:          int dayOfWeek = calendar.get(Calendar.DAY_OF_WEEK);
  69:          String amPmStr = (amPm == Calendar.AM) ? "AM" : "PM";
  70:          
  71:          g.setColor(borderColor);
  72:          g.draw3DRect(2, 10, 240, 50, true);
  73:          g.setFont(font);
  74:          g.setColor(color);
  75:   
  76:          String s = String.format("%s:%02d:%02d%s", hour, minute, second, amPmStr);
  77:          g.drawString(s, 10, 50);
  78:          
  79:          g.setColor(borderColor);
  80:          g.draw3DRect(2, 65, 240, 30, true);
  81:          g.setFont(dateFont);
  82:          g.setColor(color);
  83:          s = String.format("%s %s %s %s", day, monthName[month], year, dayName[dayOfWeek-1]);
  84:          g.drawString(s , 10, 90);
  85:      }
  86:              
  87:      private Color getColor(String colorOf){
  88:          String color = getParameter(colorOf);
  89:          if(!color.isEmpty()){
  90:              String[] rgb = color.split(",");    
  91:              int r = getColorComponent(rgb[0]);
  92:              int g = getColorComponent(rgb[1]);
  93:              int b = getColorComponent(rgb[2]);
  94:              return new Color(r, g, b);
  95:          }else{
  96:              return Color.BLACK;
  97:          }        
  98:      }
  99:      
 100:      private int getColorComponent(String componentStr){

 102:              return Integer.parseInt(componentStr);
 103:          }catch(Exception ex){
 104:              return 1;
 105:          }
 106:      }
